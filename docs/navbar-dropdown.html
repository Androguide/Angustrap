        restrict: "E"
        replace: true
        transclude: true
        scope:
            title: "@"

        template: '
        <li class="dropdown">
            <a href="#" class="dropdown-toggle" data-toggle="dropdown">{{title}} <b class="caret"></b></a>
            <ul class="dropdown-menu" data-ng-transclude></ul>
